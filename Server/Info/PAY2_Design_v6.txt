۰.  تغییرات نسخه v6  --  رفع تناقضات

۱.  اصول بنیادین این طراحی
                                       #
                                    اصل
                                  مفهوم
                                                                             ۱
                               جداسازی کامل از سیستم قدیم
تمام جداول جدید پیشوند PAY2_ دارند. هیچ Foreign Key به جداول قدیم ندارند. سیستم قدیم و جدید می‌توانند همزمان در یک دیتابیس باشند.
                                                                             ۲
                                مساعده هوشمند از حسابداری
مانده معین پرسنل در سیستم حسابداری (به‌جز اسناد حقوق) = مساعده. فرقی نمی‌کند که کارمند مساعده نقدی گرفته یا از سازمان خرید کرده.
                                                                             ۳
                                              همه چیز قابل تنظیم
هیچ مقدار ثابتی در کد وجود ندارد. تمام نرخ‌ها، مبالغ و رفتارهای محاسباتی از جدول PAY2_CONFIG خوانده می‌شوند.
                                                                             ۴
                                              آیتم‌های حکم پویا
ساختار حکم ستون ثابت ندارد. هر آیتم حقوقی جدید بدون تغییر دیتابیس قابل افزودن است.
                                                                             ۵
                                                            چند کارگاه
هر کارگاه (TAGCOD) حساب‌های حسابداری، بیمه و مالیات مستقل دارد.
                                                                             ۶
                                                        حسابرسی کامل
هر محاسبه با تمام جزئیات ذخیره می‌شود. بازتولید نتیجه هر ماه در هر زمان ممکن است.

                         💡 ایده محوری  --  مساعده هوشمند
سیستم حسابداری برای هر پرسنل یک حساب معین (HESAB) دارد. هر تراکنش مالی (مساعده نقدی، خرید از کانتین، خرید از فروشگاه سازمان) به بدهکار این حساب ثبت می‌شود. سند حقوق ماه قبل به بستانکار ثبت می‌شود. در زمان محاسبه حقوق ماه جاری، سیستم مانده این حساب را (با حذف اسناد حقوق جاری) می‌خواند و به عنوان مساعده کسر می‌کند.

۳.  گروه A  --  پیکربندی سیستم
                      ۳.۱  PAY2_CONFIG  --  جدول تنظیمات مرکزی
CREATE TABLE [dbo].[PAY2_CONFIG]
(
  [CFG_KEY]       NVARCHAR(80)  NOT NULL,  -- کلید یکتا تنظیم
  [CFG_VALUE]     NVARCHAR(500) NOT NULL,  -- مقدار جاری
  [CFG_OPTIONS]   NVARCHAR(500) NULL,      -- گزینه‌های مجاز با | (مثال '30|REAL')
  [CFG_DEFAULT]   NVARCHAR(500) NOT NULL,  -- مقدار پیش‌فرض کارخانه
  [CFG_SECTION]   NVARCHAR(60)  NOT NULL,  -- گروه در UI تنظیمات
  [LABEL_FA]      NVARCHAR(200) NOT NULL,  -- عنوان فارسی
  [DESC_FA]       NVARCHAR(1000)NULL,      -- توضیح کامل
  [OPT_LABELS]    NVARCHAR(500) NULL,      -- عنوان فارسی هر گزینه با |
  [DATA_TYPE]     NVARCHAR(20)  NOT NULL CONSTRAINT DF_CFG_DT DEFAULT('TEXT'),
  -- TEXT | INT | DECIMAL | BOOL | DATE
  [ACCESS_LEVEL]  TINYINT  NOT NULL CONSTRAINT DF_CFG_AL DEFAULT(2),
  -- 1=Super Admin | 2=Admin | 3=Payroll Manager
  [CHANGED_AT]    DATETIME NULL,
  [CHANGED_BY]    INT      NULL,
  [CHANGE_NOTE]   NVARCHAR(300) NULL,
  CONSTRAINT PK_PAY2_CONFIG PRIMARY KEY ([CFG_KEY])
);

CREATE TABLE [dbo].[PAY2_CONFIG_LOG]
(
  [LOG_ID]     INT           NOT NULL IDENTITY(1,1),
  [CFG_KEY]    NVARCHAR(80)  NOT NULL,
  [OLD_VALUE]  NVARCHAR(500) NULL,
  [NEW_VALUE]  NVARCHAR(500) NOT NULL,
  [CHANGED_BY] INT           NOT NULL,
  [CHANGED_AT] DATETIME      NOT NULL CONSTRAINT DF_CFL_DT DEFAULT(GETDATE()),
  [REASON]     NVARCHAR(300) NULL,
  CONSTRAINT PK_PAY2_CONFIG_LOG PRIMARY KEY ([LOG_ID])
);

-- Trigger لاگ خودکار
CREATE TRIGGER [dbo].[TR_PAY2_CONFIG_LOG] ON [dbo].[PAY2_CONFIG] AFTER UPDATE AS
BEGIN
  SET NOCOUNT ON;
  INSERT INTO PAY2_CONFIG_LOG(CFG_KEY,OLD_VALUE,NEW_VALUE,CHANGED_BY,CHANGED_AT,REASON)
  SELECT i.CFG_KEY,d.CFG_VALUE,i.CFG_VALUE,
         ISNULL(i.CHANGED_BY,0),GETDATE(),i.CHANGE_NOTE
  FROM INSERTED i INNER JOIN DELETED d ON i.CFG_KEY=d.CFG_KEY
  WHERE i.CFG_VALUE<>d.CFG_VALUE;
END;

                                ۵.۳  PAY2_LEAVE_BAL  --  مانده مرخصی
                                              ℹ️ مانده به دقیقه
از فرم ثبت مرخصی: مرخصی درخواستی و مانده مرخصی هر دو به صورت روز+ساعت+دقیقه نمایش داده می‌شوند. کل محاسبه به دقیقه انجام می‌شود. (۲۶ روز x ۴۴۰ دقیقه = ۱۱,۴۴۰ دقیقه استحقاق سالانه)

CREATE TABLE [dbo].[PAY2_LEAVE_BAL]
(
  [EMP_ID]          INT      NOT NULL,
  [YEAR]            SMALLINT NOT NULL,     -- سال شمسی
  -- ─── استحقاق ─────────────────────────────────────────────────
  [ENTITLEMENT_MIN] INT NOT NULL CONSTRAINT DF_LB_ENT DEFAULT(11440),
  -- استحقاق سالانه (دقیقه)  --  پیش‌فرض: ۲۶ روز x ۴۴۰ دقیقه
  -- (۴۴۰ = ۸ ساعت - جمعه + ۴ ساعت پنجشنبه --> میانگین روزانه)
  -- ─── استفاده‌شده ─────────────────────────────────────────────
  [USED_MIN]        INT NOT NULL CONSTRAINT DF_LB_USD DEFAULT(0),
  -- مجموع مرخصی‌های استفاده‌شده (دقیقه)
  -- ─── انتقالی ─────────────────────────────────────────────────
  [CARRIED_IN_MIN]  INT NOT NULL CONSTRAINT DF_LB_CIN DEFAULT(0),
  -- انتقالی از سال قبل (دقیقه)  --  MAX = 9 روز = 3960 دقیقه
  [CARRIED_OUT_MIN] INT NOT NULL CONSTRAINT DF_LB_COU DEFAULT(0),
  -- منتقل‌شده به سال بعد (دقیقه)
  -- ─── ستون‌های محاسبه‌شده (نمایشی) ────────────────────────────
  [BALANCE_MIN]     AS ([ENTITLEMENT_MIN] + [CARRIED_IN_MIN] - [USED_MIN]),
  -- مانده کل به دقیقه
  [BALANCE_DAYS]    AS (([ENTITLEMENT_MIN] + [CARRIED_IN_MIN] - [USED_MIN]) / 440),
  -- ۱ روز مرخصی = ۴۴۰ دقیقه (روز کاری ۸ ساعت کمتر از ۲۰ دقیقه وقفه)
  -- مانده به روز (تقریبی  --  برای نمایش در فیش)
  [UPDATED_AT]      DATETIME NULL,
  CONSTRAINT PK_PAY2_LEAVE_BAL PRIMARY KEY ([EMP_ID],[YEAR]),
  CONSTRAINT FK_LB_EMP FOREIGN KEY ([EMP_ID]) REFERENCES [PAY2_EMPLOYEE]([EMP_ID])
);

-- View برای مانده وام هر پرسنل
CREATE VIEW [dbo].[V_PAY2_LOAN_BALANCE] AS
SELECT
  L.EMP_ID,
  L.LOAN_ID,
  L.AMOUNT                                   AS TOTAL_AMOUNT,
  L.PAID_INST * L.INSTALLMENT               AS TOTAL_PAID,
  L.AMOUNT - L.PAID_INST * L.INSTALLMENT    AS BALANCE,
  L.INSTALLMENT                              AS NEXT_INSTALLMENT,
  (L.TOTAL_INST - L.PAID_INST)              AS REMAINING_INST
FROM PAY2_LOAN L
WHERE L.IS_ACTIVE = 1 AND L.PAID_INST < L.TOTAL_INST;

۱۰.  گروه H  --  مساعده هوشمند از حسابداری
                         💡 منطق اصلی  --  کشف از کد قدیمی
سیستم مستقیماً از DEED_HED + DEED_DTL می‌خواند. فیلتر زمانی: N_S سند < N_S سند حقوق جاری (نه نوع سند!). یعنی هر چیزی که قبل از ثبت سند حقوق روی حساب پرسنل زده شده، مساعده محسوب می‌شود  --  اعم از مساعده نقدی، خرید از فروشگاه، خرید از کانتین یا هر تراکنش دیگری.

                      ۱۰.۱  تحلیل کد مساعده سیستم قدیم
                 Query اصلی سیستم قدیم (از فرم WORKING_SUB2):

-- حالت ۱: فقط فیلتر HES_K و HES_M (بدون تفکیک تفصیلی)
SELECT SUM(D.BED - D.BES) AS MAND
FROM DEED_HED H
INNER JOIN WORKING W ON W.WDATE = H_date_filter  -- join منطقی ماه
INNER JOIN DEED_DTL D ON W.PCODE = D.HES_T        -- کد پرسنل = کد تفصیلی
                     AND H.N_S = D.N_S
WHERE D.HES_K = PSK1                               -- کد کل حساب (از تنظیمات)
  AND D.HES_M = PSM1                               -- کد معین حساب (از تنظیمات)
  AND H.N_S < [N_S_سند_حقوق_جاری]                -- اسناد قبل از سند حقوق این ماه
  AND WORKING.WDATE = [ماه_جاری]
  AND Umonth(H.DATE_S) = Umonth([ماه_جاری])       -- تاریخ سند در همان ماه

-- حالت ۲: با فیلتر اضافی HES_T (تفصیلی مشخص)
-- ... همان query + AND D.HES_T = PST1

                     ۱۰.۲  جدول ساختار حسابداری مرتبط
                               جدول/ستون
                           در سیستم قدیم
                     نقش در محاسبه مساعده
                                                                   DEED_HED.N_S
                                                      شماره سند (float)
              کلید اتصال به DEED_DTL و فیلتر < N_S حقوق
                                                                DEED_HED.DATE_S
                                                     تاریخ سند (bigint)
                            فیلتر ماه: Umonth(DATE_S) = ماه حقوق
                                                                  DEED_HED.NO_S
                                                          سری سند (float)
برای تفکیک سری‌های مختلف اسناد (در صورت نیاز)
                                                                 DEED_DTL.HES_K
                                                       کد کل حساب (int)
     = PSK1 از تنظیمات  --  سرفصل دریافتنی از پرسنل
                                                                 DEED_DTL.HES_M
                                                   کد معین حساب (int)
                    = PSM1 از تنظیمات  --  معین حساب پرسنل
                                                                 DEED_DTL.HES_T
                                                        کد تفصیلی (int)
                                       = PCODE پرسنل = PAY2_EMPLOYEE.ACC_T
                                                              DEED_DTL.HES_T2-4
                                               تفصیلی‌های اضافه
سطوح بیشتر  --  معمولاً در مساعده استفاده نمی‌شود
                                                                   DEED_DTL.BED
                                                           بدهکار (float)
   مبالغ بدهکار حساب پرسنل (مساعده داده‌شده)
                                                                   DEED_DTL.BES
                                                       بستانکار (float)
           مبالغ بستانکار (تسویه از حقوق ماه قبل)
                                                                   DEED_DTL.HES
                                              کد معین متنی (nvarchar)
                   Index جستجوی سریع  --  ترکیب HES_K+HES_M+HES_T

۱۰.۴  V_PAY2_ADVANCE_BALANCE  --  View مساعده هوشمند (نهایی)
                                     ✅ این View نهایی‌شده است
بر اساس ساختار واقعی DEED_HED و DEED_DTL طراحی شده. کدهای HES_K و HES_M از PAY2_CONFIG خوانده می‌شوند. کد تفصیلی پرسنل از PAY2_EMPLOYEE.ACC_T.

-- ================================================================
-- V_PAY2_ADVANCE_BALANCE
-- مانده مساعده هر پرسنل برای یک ماه و یک سند حقوق خاص
-- پارامتر: @PERIOD_DATE (ماه)، @PAYROLL_N_S (شماره سند حقوق)
-- ================================================================

-- تابع کمکی: تبدیل تاریخ شمسی به ماه (مشابه Umonth سیستم قدیم)
CREATE FUNCTION [dbo].[FN_PAY2_MONTH](@DATE BIGINT)
RETURNS INT AS
BEGIN
  RETURN @DATE / 100  -- YYYYMM
END;

-- Stored Procedure محاسبه مساعده همه پرسنل یک ماه
CREATE PROCEDURE [dbo].[SP_PAY2_GET_ADVANCES]
  @PERIOD_DATE  BIGINT,    -- ماه شمسی: مثال 14040100
  @PAYROLL_N_S  FLOAT,     -- شماره سند حقوق این ماه در DEED_HED
  @WS_ID        INT        -- کارگاه
AS
BEGIN
  SET NOCOUNT ON;

  -- v6: خواندن کدهای حساب از PAY2_WORKSHOP_ACC (کارگاه‌محور  --  نه global)
  DECLARE @HES_K   INT = (SELECT CAST(ACC_CODE AS INT) FROM PAY2_WORKSHOP_ACC
                           WHERE WS_ID=@WS_ID AND ACC_KEY='ADV_HES_K')
  DECLARE @HES_M   INT = (SELECT CAST(ACC_CODE AS INT) FROM PAY2_WORKSHOP_ACC
                           WHERE WS_ID=@WS_ID AND ACC_KEY='ADV_HES_M')
  DECLARE @USE_T   BIT = CAST((SELECT CFG_VALUE FROM PAY2_CONFIG WHERE CFG_KEY='ADV_USE_HES_T_FILTER') AS BIT)
  DECLARE @MIN_POS BIT = CAST((SELECT CFG_VALUE FROM PAY2_CONFIG WHERE CFG_KEY='ADV_MIN_POSITIVE') AS BIT)
  -- v6: ADV_SCOPE  --  محدوده جستجو: CURRENT_MONTH=فقط همین ماه | OPEN_BALANCE=کل مانده باز
  DECLARE @ADV_SCOPE NVARCHAR(20) = ISNULL(
    (SELECT CFG_VALUE FROM PAY2_CONFIG WHERE CFG_KEY='ADV_SCOPE'), 'CURRENT_MONTH')
  DECLARE @PERIOD_MONTH INT = @PERIOD_DATE / 100  -- YYYYMM

  -- بررسی پیکربندی کارگاه
  IF @HES_K IS NULL OR @HES_M IS NULL
  BEGIN
    RAISERROR(N'PAY2_WORKSHOP_ACC: ADV_HES_K یا ADV_HES_M برای کارگاه %d تنظیم نشده است.',16,1,@WS_ID)
    RETURN
  END

  SELECT
    E.EMP_ID,
    E.ACC_T                            AS PCODE,
    E.LAST_NAME + N' ' + E.FIRST_NAME AS FULL_NAME,

    -- مانده حساب معین پرسنل از سیستم حسابداری
    ISNULL((
      SELECT CAST(SUM(D.BED - D.BES) AS BIGINT)
      FROM DEED_HED  H
      INNER JOIN DEED_DTL D ON H.N_S = D.N_S
      WHERE D.HES_K = @HES_K                        -- کد کل
        AND D.HES_M = @HES_M                        -- کد معین
        AND (
              @USE_T = 0                            -- اگر تفصیلی لازم نیست
              OR D.HES_T = E.ACC_T                  -- کد تفصیلی = کد پرسنل
            )
        AND H.N_S < @PAYROLL_N_S                    -- *** قبل از سند حقوق جاری ***
        AND (
              @ADV_SCOPE = 'OPEN_BALANCE'                       -- کل مانده باز تا سند حقوق
              OR [dbo].[FN_PAY2_MONTH](H.DATE_S) = @PERIOD_MONTH  -- فقط همان ماه
            )
        AND H.OKF = 1                               -- فقط اسناد تأیید شده
    ), 0)                                           AS RAW_BALANCE,

    -- کسر استثناهای دستی
    ISNULL((
      SELECT SUM(EXCL_AMOUNT)
      FROM PAY2_ADVANCE_EXCL
      WHERE EMP_ID = E.EMP_ID
        AND PERIOD_DATE / 100 = @PERIOD_MONTH
    ), 0)                                           AS MANUAL_EXCL,

    -- مانده نهایی مساعده
    CASE
      WHEN @MIN_POS = 1 AND
           ISNULL((
             SELECT CAST(SUM(D.BED - D.BES) AS BIGINT)
             FROM DEED_HED H INNER JOIN DEED_DTL D ON H.N_S=D.N_S
             WHERE D.HES_K=@HES_K AND D.HES_M=@HES_M
               AND (@USE_T=0 OR D.HES_T=E.ACC_T)
               AND H.N_S < @PAYROLL_N_S
               AND [dbo].[FN_PAY2_MONTH](H.DATE_S)=@PERIOD_MONTH
               AND H.OKF=1
           ),0)
           - ISNULL((SELECT SUM(EXCL_AMOUNT) FROM PAY2_ADVANCE_EXCL
                     WHERE EMP_ID=E.EMP_ID AND PERIOD_DATE/100=@PERIOD_MONTH),0)
           <= 0
      THEN 0
      ELSE
        ISNULL((
          SELECT CAST(SUM(D.BED - D.BES) AS BIGINT)
          FROM DEED_HED H INNER JOIN DEED_DTL D ON H.N_S=D.N_S
          WHERE D.HES_K=@HES_K AND D.HES_M=@HES_M
            AND (@USE_T=0 OR D.HES_T=E.ACC_T)
            AND H.N_S < @PAYROLL_N_S
            AND [dbo].[FN_PAY2_MONTH](H.DATE_S)=@PERIOD_MONTH
            AND H.OKF=1
        ),0)
        - ISNULL((SELECT SUM(EXCL_AMOUNT) FROM PAY2_ADVANCE_EXCL
                  WHERE EMP_ID=E.EMP_ID AND PERIOD_DATE/100=@PERIOD_MONTH),0)
    END                                             AS ADVANCE_DEDUCTION

  FROM PAY2_EMPLOYEE E
  INNER JOIN PAY2_PERIOD P ON P.WS_ID = E.WS_ID AND P.PERIOD_DATE/100 = @PERIOD_MONTH
  WHERE E.WS_ID = @WS_ID
    AND E.IS_ACTIVE = 1
    AND E.ACC_T IS NOT NULL;  -- فقط پرسنلی که کد حسابداری دارند
END;

CREATE FUNCTION [dbo].[FN_PAY2_CALC_TAX](@ANNUAL_BASE BIGINT, @TAX_YEAR SMALLINT)
RETURNS BIGINT AS
BEGIN
  DECLARE @TAX BIGINT = 0, @PREV_LIMIT BIGINT = 0
  DECLARE @RATE DECIMAL(5,2), @LIMIT BIGINT, @FIXED BIGINT

  DECLARE cur CURSOR FOR
    SELECT UPPER_LIMIT, RATE_PCT, FIXED_TAX FROM PAY2_TAX_BRACKET
    WHERE TAX_YEAR=@TAX_YEAR ORDER BY SORT_ORDER

  OPEN cur
  FETCH NEXT FROM cur INTO @LIMIT, @RATE, @FIXED

  WHILE @@FETCH_STATUS=0
  BEGIN
    IF @ANNUAL_BASE <= @LIMIT
    BEGIN
      SET @TAX = @FIXED + CAST((@ANNUAL_BASE - @PREV_LIMIT) * @RATE / 100 AS BIGINT)
      BREAK
    END
    SET @PREV_LIMIT = @LIMIT
    FETCH NEXT FROM cur INTO @LIMIT, @RATE, @FIXED
  END

  -- اگر از همه پله‌ها بیشتر بود: پله آخر اعمال شود
  IF @@FETCH_STATUS<>0 AND @TAX=0
    SET @TAX = @FIXED + CAST((@ANNUAL_BASE-@PREV_LIMIT)*@RATE/100 AS BIGINT)

  CLOSE cur; DEALLOCATE cur
  RETURN @TAX  -- مالیات سالانه  --  موتور /12 می‌کند
END

-- آیتم‌هایی مانند خواربار، حق اولاد، حق بن
IF @DAYSB >= 28
    SET @AMOUNT = @MONTHLY_VALUE          -- مبلغ کامل
ELSE
    SET @AMOUNT = ROUND(@MONTHLY_VALUE / 30.0 * @DAYSB)  -- تسهیم بر اساس ۳۰
نکته: عدد مبنا ۳۰ است، نه تعداد واقعی روزهای ماه  --  این رفتار سیستم قدیمی است و در PAY2 نیز حفظ می‌شود.
  ۱۵-۴.  حق نهار  --  قانون خاص (کسر جمعه و مرخصی)
-- روزهای واجد حق نهار = کارکرد - جمعه‌های ماه - روزهای مرخصی + تعطیلات رسمی
-- FRID = تعداد جمعه‌های ماه (از WORKHEAD)
-- MORAKH = مجموع روزهای مرخصی / 440 دقیقه (روزانه‌سازی)
-- TDAYS = تعطیلات رسمی‌ای که روز کاری هستند

SET @BON_DAYS = @DAYSB - @FRID - @MORAKH + @TDAYS
IF @BON_DAYS > 0
    SET @NAHAR_AMT = @BON * @BON_DAYS
ELSE
    SET @NAHAR_AMT = @BON * @DAYSB  -- fallback: پرداخت کامل کارکرد

                  ۱۵-۵.  پایه بیمه (MABMASH)  --  فرمول کامل
پایه بیمه بر اساس DAYS (اسمی) محاسبه می‌شود و آیتم‌های زیر را شامل می‌شود:MABMASH =
    (SALARY_DAYLY + JAZBx[H] + BONx[H] + SAYERx[H]
     + CONDITIONSx[H] + HAGHTAHOLx[H]) x DAYS
  + EZAFAHx[H]
  + PADASHx[H]
  + HOME   x (IF DAYS>=28: 1  ELSE DAYS/30) x [H]
  + CHILDREN x (IF DAYS>=28: 1 ELSE DAYS/30) x [H]
  + BONJ   x (IF DAYS>=28: 1  ELSE DAYS/30) x [H]
  + حق شیفت: (HSHIFT% / 100) x SALARY_DAYLYB x DAYS x [H]

-- [H] = فلگ مشمولیت از BASEKNOW (قابل تنظیم در PAY2_CONFIG)

-- حق نهار (BON) مشمول بیمه نیست
-- حق بن (BONJ) مشمول بیمه است ولی معاف مالیات

           ۱۵-۶.  دو حالت بیمه کارفرما  --  SANAD و SANAD10
کارگاه‌هایی که شامل "تبصره ماده ۷" قانون تامین اجتماعی هستند، محاسبه بیمه کارفرما متفاوتی دارند. این تفاوت از طریق فیلد INS_MODE در PAY2_WORKSHOP کنترل می‌شود.
               حالت ۱  --  کارگاه معمولی (INS_MODE = 1 / SANAD)
-- H7BIM  = جمع پایه کارگران عادی (TAB56=0, DIHARD=0, BIMEH_NUM>50)
-- H3BIM  = جمع پایه مدیران (ISCEO=True)  --  از 3% بیکاری معاف
-- H7BIMJANB = جمع پایه جانبازان (DIHARD=True)  --  نرخ 18%

bimah10 =
    Round(H7BIM x 0.20)              -- ۲۰% کارفرما
  + Round((H7BIM - H3BIM) x 0.03)   -- ۳% بیکاری (غیر مدیر)
  + Round(H7BIMJANB x 0.18)          -- ۱۸% جانباز (بدون ۳% بیکاری)

        حالت ۲  --  کارگاه تبصره‌ای (INS_MODE = 2 / SANAD10)
در این حالت برای کارگاه‌هایی که تعداد کارگران مشمول آن‌ها بیشتر از ۵ نفر باشد، ۲۰% کارفرما فقط برای مازاد ۵ نفر اول لازم است:-- EPERS = تعداد کارگران مشمول
--   (DIHARD=0 AND BIMEH_NUM>50 AND TAB56=0 AND DAYS>0)
-- H7BIM3 = جمع پایه همه مشمولان شامل TAB56  --  برای 3% بیکاری
-- H7BIM  = جمع پایه کارگران عادی (TAB56=0)  --  برای 20%

IF EPERS > 5
    bimah10 =
        Round(H7BIM3 x 0.03)                             -- ۳% بیکاری (همه)
      + Round(H7BIM / EPERS x 0.20 x (EPERS - 5))        -- ۲۰% فقط برای مازاد ۵ نفر
      + Round(H7BIMJANB x 0.18)                           -- ۱۸% جانباز
ELSE
    bimah10 =
        Round(H7BIM3 x 0.03)                             -- ۳% بیکاری (همه)
      + Round(H7BIMJANB x 0.18)                           -- ۱۸% جانباز
      -- ۲۰% کارفرما صفر است (کمتر از ۵ نفر = معافیت کامل)
